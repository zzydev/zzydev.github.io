[原题链接](https://leetcode.cn/problems/remove-element/)

```cpp
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int k = 0;
        for (int i = 0; i < nums.size(); i ++) {
            if (nums[i] != val) {
                nums[k ++] = nums[i];
            }
        }
        return k;
    }
};
```
