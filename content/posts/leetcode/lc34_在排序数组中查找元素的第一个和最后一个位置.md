[原题链接](https://leetcode.cn/problems/find-first-and-last-position-of-element-in-sorted-array/)

模板题：

```cpp
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int n = nums.size();
        int l = lowerBound(nums, n, target);
        int r = upperBound(nums, n, target);
       // 特判 [] 0 和 target不在nums中的情况
        if (l == nums.size() || nums[l] != target) return {-1, -1};
        return {l, r - 1};
    }

    int lowerBound(vector<int>& A, int n, int target) {
        int l = 0, r = n;
        while (l < r) {
            const int mid = l + ((r - l) >> 1);
            if (A[mid] < target) {
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        return l;
    }

    int upperBound(vector<int>& A, int n, int target) {
        int l = 0, r = n;
        while(l < r) {
            const int mid = l + ((r - l) >> 1);
            if (A[mid] <= target) {
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        return l;
    }
};
```
