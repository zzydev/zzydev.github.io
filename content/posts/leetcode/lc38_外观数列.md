[原题链接](https://leetcode.cn/problems/count-and-say/)

```cpp
class Solution {
public:
    string countAndSay(int n) {
        string res = "1";
        for (int i = 0 ;i < n - 1; i ++) {
            string t = "";
            for (int j = 0; j < res.size();) {
                int k = j + 1;
                while (k < res.size() && res[k] == res[j]) k ++;
                t += to_string(k - j) + res[j];
                j = k;
            }
            res = t;
        }
        return res;
    }
};
```
