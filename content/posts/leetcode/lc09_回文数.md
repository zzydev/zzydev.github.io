### 解法一

```cpp
class Solution {
public:
    bool isPalindrome(int x) {
        string t = to_string(x);
        auto res = t;
        reverse(t.begin(), t.end());
        return res == t;
    }
};
```

### 解法二

如果负数直接返回 false， 比较正序逆序数字是否一样

```cpp
class Solution {
public:
    bool isPalindrome(int x) {
        if (x < 0) return false;
        long long t = 0;
        int ans = x;
        while (x) {
            t = t * 10 + x % 10;
            x /= 10;
        }
        return ans == t;
    }
};
```
