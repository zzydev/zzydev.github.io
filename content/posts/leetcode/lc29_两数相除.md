```cpp
class Solution {
public:
    int divide(int dividend, int divisor) {
        typedef long long LL;
        vector<LL> exp;
        LL res = 0;
        bool is_minus = false;
        if (dividend > 0 ^ divisor > 0) is_minus = true;
        LL x = abs((LL) dividend), y = abs((LL) divisor);

        // 2 ^ 1  2 ^ 2  2 ^ 3  ... 
        for (LL i = y; i <= x; i = i  + i) exp.push_back(i); 

        for (int i = exp.size() - 1; i >= 0; i --) {
            if (x >= exp[i]) {
                x -= exp[i];
                res += 1ll << i;
            }
        }
        if (is_minus) res = -res;
        if (res > INT_MAX || res < INT_MIN) res = INT_MAX;
        return res;
    }
};
```
