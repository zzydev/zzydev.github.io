```cpp
class Solution {
public:
    int maxArea(vector<int>& height) {
        int len  = height.size();
        int res = 0;
        for (int l = 0, r = len - 1; l < r;) {
            res = max(res, min(height[l], height[r]) * (r - l));
            if (height[l] >= height[r]) r --;
            else if (height[l] < height[r]) l ++;
        }
        return res;
    }
};
```
